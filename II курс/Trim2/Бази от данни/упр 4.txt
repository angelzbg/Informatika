имат цена >=500 и <= 1000, подредете резултата по цена

SELECT NAME, PRICE 
FROM PRODUCTS 
--WHERE PRICE >= 500 AND PRICE <= 1000 
WHERE PRICE BETWEEN 500 AND 1000

/*Изведете наименованията на длъжностите с минимална 
заплата > 5000 и максимална заплата > 8000, подредете 
резултата по минимална и максимална заплата възходящо*/ 
SELECT JOB_TITLE, MIN_SALARY, MAX_SALARY 
FROM JOBS 
WHERE MIN_SALARY > 5000 AND MAX_SALARY > 8000 
ORDER BY 2,3

/*Изведете стойността на поръчка 2355*/ 
SELECT SUM(UNIT_PRICE*QUANTITY) AS TOTAL 
FROM ORDER_ITEMS

/*Изведете средната месечна заплата в отдел 80*/ 
SELECT AVG(SALARY/12) 
FROM EMPLOYEES 
WHERE DEPARTMENT_ID = 80 
/*Изведете най-ниската и най-високата заплата в 
отдел 50*/ 
SELECT MIN(SALARY) AS MIN_SALARY, MAX(SALARY) AS MAX_SALARY 
FROM EMPLOYEES 
WHERE DEPARTMENT_ID = 50 
/*Изведете средната месечна заплата и номера на отдела, 
в която се дава тази заплата*/ 
SELECT DEPARTMENT_ID, AVG(SALARY/12) 
FROM EMPLOYEES 
GROUP BY DEPARTMENT_ID 

/*Изведете номерата на продуктите и общото количество, 
в което са били поръчвани*/ 
SELECT PRODUCT_ID, SUM(QUANTITY) AS QTY 
FROM ORDER_ITEMS 
GROUP BY PRODUCT_ID 
ORDER BY 1
