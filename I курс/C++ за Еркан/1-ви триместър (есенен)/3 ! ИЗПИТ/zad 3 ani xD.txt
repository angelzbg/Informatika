/*
Някъв бавняр тича всеки ден в продължение на 15 дни.
Да се напише програма, която да въвежда изминатите километри за всеки ден;
да извежда най-голямото изминато разстояние за ден и дните, в които е изминато;
да изведе сумата на изминатото разстояние за всички дни.
*/


#include <iostream>
using namespace std;
int main()
{
	system("chcp 1251");
	const short n = 15;
	float a[n], Sum1 = 0;

	for (short i = 0; i < n; i++)
	{
		cout << "Колко бяга бавняра в ден: " << i + 1 << " , за да не е Фики и да има какички: "; cin >> a[i];
		Sum1 += a[i];
	}

	float max = a[0];
	for (short i = 1; i < n; i++)
	{
		if (a[i]>max)
		{
			max = a[i];
		}
	} cout << "Бавняра счупи крачка на " << max << " км. в ден: ";

	for (short i = 0; i < n; i++)
	{
		if (a[i] == max)
		{
			cout << i + 1 << " ";
		}
	}  cout << endl;

	cout << "Успя да изтича: " << Sum1 << " km. Повече другия път!" << endl;

	system("pause");
	return 0;
}
