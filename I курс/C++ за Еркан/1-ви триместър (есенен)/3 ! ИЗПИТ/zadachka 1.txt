/*
1. Да се въведат количествата събрани ягоди за всеки ден от 15 до 30 май.
2. Да се намери, през кои два поредни дни има общо най-много събрани плодове.
3. Да се въведе число и да се изведат дните със събрано количество ягоди повече от въведеното число.
4. Да се изведе сумата от всички количества.
*/

// ползвам 'i+15' при cout, защото в тъпото задание искаме от 15ти до 30ти май, но масива започва от 0 и свъшва на 15тия елемент, т.е.
// трябва да изкарам индекса+15, но самия вход на стойност към елемент си остава с оригинален индекс 'i'
#include <iostream>
using namespace std;

int main()
{
 	// точка 1
	const short n = 16; //задаваме за 'n' да е със стойност 16, която не може да се измени
	int a[n]; //създаваме масив 'a' от тип цели числа с брой 'n' елемента, т.е. масив със 16 елемента

	for (short i = 0; i < n; i++) //с този цикъл се обхождат всички елементи на масива
	{
		cout << "Kolichestvo qgodi za " << i + 15 << " May : "; // извежда като начало "Kolichestvo qgodi za 15 May :" , "Kolichestvo qgodi za 16 May :" и т.н.
		cin >> a[i]; // приема стойност от клавиатурата за текущия елемент елемент
	}
 	// точка 2
	int MAXsum = a[0] + a[1]; // задаваме за MAXsum да е равно на първия + втория елемент на масива
	short day1 = 0, day2 = 1; // ще ги използваме за да им присвоим стойността на двата елемента, които имат общо най-голяма сума
	for (short i = 1; i < n - 1; i++) // обхожда целия масив без първия и последния елемент ( 1 и n-1 )
	{
		if (a[i] + a[i + 1]>MAXsum) // проверяваме дали текущия елемент+следващия елемент са по-големи от MAXsum
		{ // в тези скоби се изпълнява само ако горното условие е изпълнено
			MAXsum = a[i] + a[i + 1]; // присвояваме за MAXsum стойността на текущия елемент+следващия елемент
			day1 = i; // присвояваме на day1 индекса на текущия елемент
			day2 = i + 1; // присвояваме на day2 стойността на елемента след текущия елемент
		}
	}
	cout << "Naj-mnogo iagodi v posledovatelni dni ima v dnite " << day1 + 15 << " & " << day2 + 15 << " prez mesec May = " << MAXsum << endl;

 	//точка 3
	float chislo; // chislo може да е дробно или цяло с float
	cout << "Vudevede chislo: ";
	cin >> chislo;
	cout << "Dni sus subrano kolichestvo pvoeche ot " << chislo << " sa: ";
	for (short i = 0; i < n; i++) // обхождаме целия масив
	{
		if (a[i]>chislo) // ако текущия елемент е по-голям от въведеното число
		{
			cout << i + 15 << " "; // принтва индекса на елементите по-големи от chislo с добавена стойност 15
		}
	} cout << endl;

	int sum = 0; // задаваме за sum да е от цели числа и да е равно на 0
	for (short i = 0; i < n ; i++) // обхождаме целия масив
	{
		sum = sum + a[i]; // sum е равно на предишната стойност за sum + текущия елемент
	}
	cout << "Sumata na vsichki subrani qgodi prez mesec May = " << sum << endl;

	system("pause");
	return 0;
}
